60      /**
61       * Perform all the strategy resolution and return a list of fully
62       * initialized players.
63       * 
64       * @return The unmodifiable collection of players, in a totally randomized
65       *         order.
66       */
67      public Collection<Player> assemblePlayers() {
68          // parse a list of players
69          final Map<String, String> playerStrategies = new HashMap<>();
70          final Map<String, URI> strategyJars = new HashMap<>();
71          for (final String playerName : this.config.stringPropertyNames()) {
72              final String strategyDescr = this.config.getProperty(playerName);
73              final String[] parts = strategyDescr.split("\\Q@\\E");
74              if (parts.length != 2) {
75                  throw new IllegalArgumentException(
76                          "Invalid strategy descriptor: " + strategyDescr);
77              }
78              final String strategyClass = parts[0];
79              if (strategyClass.startsWith("org.drooms.impl.")) {
80                  throw new IllegalStateException(
81                          "Strategy musn't belong to the game implementation package.");
82              }
83              try {
84                  final URI strategyJar = new URI(parts[1]);
85                  playerStrategies.put(playerName, strategyClass);
86                  strategyJars.put(strategyClass, strategyJar);
87              } catch (final URISyntaxException e) {
88                  throw new IllegalArgumentException(
89                          "Invalid URL in the strategy descriptor: "
90                                  + strategyDescr, e);
91              }
92          }
93          // load strategies for players
94          final List<Player> players = new ArrayList<>();
95          for (final Map.Entry<String, String> entry : playerStrategies
96                  .entrySet()) {
97              final String playerName = entry.getKey();
98              final String strategyClass = entry.getValue();
99              final URI strategyJar = strategyJars.get(strategyClass);
100             Strategy strategy;
101             try {
102                 strategy = this.loadStrategy(strategyClass, strategyJar);
103             } catch (final Exception e) {
104                 throw new IllegalArgumentException("Failed loading: "
105                         + strategyClass, e);
106             }
107             final KnowledgeBuilder kb = strategy.getKnowledgeBuilder(this
108                     .loadJar(strategyJar));
109             try {
110                 final KnowledgeBase kbase = kb.newKnowledgeBase();
111                 players.add(new Player(playerName, kbase));
112             } catch (final Exception ex) {
113                 for (final KnowledgeBuilderError error : kb.getErrors()) {
114                     PlayerAssembly.LOGGER.error(error.toString());
115                 }
116                 throw new IllegalStateException(
117                         "Cannot create knowledge base for strategy: "
118                                 + strategy.getName(), ex);
119             }
120         }
121         Collections.shuffle(players, new SecureRandom());
122         return Collections.unmodifiableList(players);
123     }

125     /**
126      * Load a strategy JAR file in its own class-loader, effectively isolating
127      * the strategies from each other.
128      * 
129      * @param strategyJar
130      *            The JAR coming from the player config.
131      * @return The class-loader used to load the strategy jar.
132      */
133     private ClassLoader loadJar(final URI strategyJar) {
134         if (!this.strategyClassloaders.containsKey(strategyJar)) {
135             @SuppressWarnings("resource")
136             final ClassLoader loader = URLClassLoader.newInstance(
137                     new URL[] { PlayerAssembly.uriToUrl(strategyJar) }, this
138                             .getClass().getClassLoader());
139             this.strategyClassloaders.put(strategyJar, loader);
140         }
141         return this.strategyClassloaders.get(strategyJar);
142     }
143 
144     private Strategy loadStrategy(final String strategyClass,
145             final URI strategyJar) throws Exception {
146         if (!this.strategyInstances.containsKey(strategyClass)) {
147             final Class<?> clz = Class.forName(strategyClass, true,
148                     this.loadJar(strategyJar));
149             final Strategy strategy = (Strategy) clz.newInstance();
150             this.strategyInstances.put(strategyClass, strategy);
151         }
152         return this.strategyInstances.get(strategyClass);
153     }
